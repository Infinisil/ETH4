#+BEGIN_SRC dot :file ./hi.png :cmdline -Kdot -Tpng :exports results
graph {
  subgraph entities {
    node [shape=box];
    Student;
    Lecture;
  }
  subgraph relations {
    node [shape=diamond];
    attends;
  }
  { Legi, Name, Semester } -- Student;
  Student -- attends [label=Attendant];
  attends -- Lecture [label=Course];
  Lecture -- { Nr, Title, CP };
}
